import pandas as pd
import matplotlib.pyplot as plt

# Assuming you have loaded your dataset into a DataFrame named 'df'
df = pd.read_csv('IMDB.csv')  

# Group by 'Year' and calculate the count of movies in each year
movies_per_year = df.groupby('Year of relase')['Name of movie'].count()

# Plotting the graph
plt.figure(figsize=(12, 6))
plt.bar(movies_per_year.index, movies_per_year, color='skyblue')
plt.title('Number of Movies Released Each Year')
plt.xlabel('Year')
plt.ylabel('Number of Movies')
plt.show()
